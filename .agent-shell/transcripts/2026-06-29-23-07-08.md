```python
def create_combined_dataset():
    # Load datasets
    cavitte_ds = xr.open_zarr('Cavitte_2020/cavitte_2020_consolidated.zarr')
    chung_ds = xr.open_zarr('Chung_2023/chung_2023_consolidated.zarr')

    # Add dataset dimension
    cavitte_ds = cavitte_ds.expand_dims('dataset').assign_coords(dataset=['Cavitte_2020'])
    chung_ds = chung_ds.expand_dims('dataset').assign_coords(dataset=['Chung_2023'])

    # Create combined age coordinate
    combined_ages = sorted(set(cavitte_ds.age.values) | set(chung_ds.age.values))

    # Reindex both datasets to combined ages (fills missing with NaN)
    cavitte_ds = cavitte_ds.reindex(age=combined_ages)
    chung_ds = chung_ds.reindex(age=combined_ages)

    # Concatenate along dataset dimension
    combined_ds = xr.concat([cavitte_ds, chung_ds], dim='dataset')

    # Save combined dataset
    combined_ds.to_zarr('combined_isochrones.zarr')

    return combined_ds
```

```python
# Step 2: Smart query function
def query_isochrones(metadata_df, age_range, variables=None, datasets=None):
    # Filter metadata
    query_mask = (metadata_df['age'] >= age_range[0]) & (metadata_df['age'] <= age_range[1])
    if datasets:
        query_mask &= metadata_df['dataset'].isin(datasets)
    if variables:
        query_mask &= metadata_df['variables'].apply(lambda x: any(v in x for v in variables))

    filtered_meta = metadata_df[query_mask]

    # Load only the needed data chunks
    results = {}
    for _, row in filtered_meta.iterrows():
        ds = xr.open_zarr(row['zarr_path'])
        flight_data = ds.sel(flight_line=row['flight_line'], age=row['age'])
        if row['dataset'] not in results:
            results[row['dataset']] = []
        results[row['dataset']].append(flight_data)

    # Combine results by dataset
    return {name: xr.concat(data_list, dim='flight_line')
```
